# Step 13: 前端沉浸式创作平台重构 — TypeScript全栈升级与用户体验革新

> 版本: 1.0
> 日期: 2026-04-29
> 依赖: Step1-12 (全栈)
> 目标: 从"功能可用的管理面板"升级为"沉浸式AI小说创作平台"
> 核心转变: 从后端能力映射 → 到用户创作旅程驱动

---

## 1. 设计哲学

### 1.1 核心转变

```
从：后端功能的前端映射面板                    → 到：以创作者为中心的沉浸式工作空间
从：Element Plus标准组件                      → 到：自研设计系统 + 行业级交互范式
从：3秒轮询刷新任务状态                        → 到：WebSocket实时流 + 乐观更新
从：表格列表展示数据                          → 到：可视化图谱 + 空间化信息架构
从：单页面独立功能                            → 到：上下文感知的统一工作空间
从：静态配置表单                              → 到：AI驱动的智能配置向导
从：简单的文本日志                            → 到：结构化日志中心 + 智能诊断
```

### 1.2 以用户为中心的设计原则

**原则1: 创作流 uninterrupted**
创作者不应被技术细节打断。所有AI操作、任务调度、系统配置都应融入创作流，而非跳出创作环境。

**原则2: 上下文即一切**
界面始终感知用户当前在做什么（写大纲/设计角色/生成章节），并提供上下文相关的AI建议和工具。

**原则3: 可视化即理解**
复杂信息（角色关系、情节结构、世界规则、Agent执行链）必须通过可视化呈现，降低认知负荷。

**原则4: 实时即可信**
所有状态变更即时反馈。任务进度、Agent思考、系统健康，用户无需刷新即可看到最新状态。

**原则5: 渐进式能力披露**
新手看到简洁界面，专家通过快捷键/命令面板/高级模式访问全部能力。禁止一次性展示所有功能。

**原则6: 键盘优先，鼠标可选**
所有高频操作支持键盘快捷键。鼠标用于探索和可视化交互，键盘用于高效操作。

## 3. 技术栈决策

### 3.1 技术栈选型（Vue3保留升级路线）

经过深度评估，选择 **保留Vue3 + TypeScript** 并全面升级，而非迁移到React：

**保留Vue3的理由**:
1. 现有代码投资：约2000行Vue3+TS代码，迁移成本 > 2周
2. Vue3 Composition API 已足够现代化，功能对标React Hooks
3. Vue3的性能（响应式系统、编译优化）在2024年已不逊色于React
4. 团队熟悉度：现有团队使用Vue3开发，无需学习成本

**全面升级内容**:

| 层级 | 当前 | Step13升级 |
|------|------|-----------|
| 框架 | Vue 3.4 | **Vue 3.4** (保留，启用 `<script setup>` + 组合式函数) |
| 构建 | Vite 5 | **Vite 5** (保留，启用 SSR-friendly 构建) |
| 语言 | TypeScript 5.5 | **TypeScript 5.5** (启用 strict 模式 + noImplicitAny) |
| UI组件 | Element Plus 2.7 | **自定义设计系统** + shadcn-vue + Radix Vue primitives |
| 样式 | TailwindCSS 3 | **TailwindCSS 3** (保留，扩展设计令牌) |
| 状态 | Pinia 2.1 | **Pinia 2.1** (保留，增加ORM层 + 持久化插件) |
| 路由 | Vue Router 4 | **Vue Router 4** (保留，增加路由守卫 + 懒加载) |
| 可视化 | ECharts 5.5 | **ECharts 5.5** + **Vue Flow** + **D3.js 7** + **Three.js** (3D场景) |
| 动画 | CSS动画 | **GSAP 3** + **@vueuse/motion** + CSS动画 |
| 实时通信 | 无 | **原生WebSocket** + **EventSource(SSE)** 双通道 |
| 编辑器 | 无 | **TipTap 2** (ProseMirror封装，块级编辑器) |
| 图表 | ECharts | **ECharts** + **Mermaid** (文本生成图表) |
| 类型验证 | 无 | **Zod 3** (运行时类型验证，API响应守卫) |
| 虚拟滚动 | 无 | **vue-virtual-scroller** (大数据列表) |
| 拖拽 | 无 | **VueDraggablePlus** (拖拽排序) |
| 命令面板 | 无 | **@vueuse/integrations** + 自定义命令面板 |
| 图标 | Element Icons | **Lucide Vue** (更现代、一致的图标) |
| 日期 | date-fns | **date-fns** (保留) + **@vueuse/date** |
| HTTP | Axios | **Axios** (保留，增加请求去重 + 自动重试) |
| 工具库 | 无 | **VueUse 10** (全面使用) |

// stores/novel.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { z } from 'zod'
import { useApi } from '@/composables/useApi'
import { useSocket } from '@/composables/useSocket'

const NovelSchema = z.object({
  id: z.string(),
  title: z.string(),
  genre: z.enum(['xianxia', 'wuxia', 'sci-fi', 'fantasy', 'modern', 'horror', 'history']),
  progress: z.number().min(0).max(100),
  status: z.enum(['draft', 'writing', 'reviewing', 'completed']),
  wordCount: z.number(),
  chapterCount: z.number(),
  lastEditedAt: z.string().datetime(),
})

type Novel = z.infer<typeof NovelSchema>

export const useNovelStore = defineStore('novel', () => {
  // State
  const novels = ref<Novel[]>([])
  const currentNovelId = ref<string | null>(null)
  const loading = ref(false)
  const error = ref<string | null>(null)

  // API
  const api = useApi()
  const socket = useSocket()

  // Getters
  const currentNovel = computed(() =>
    novels.value.find(n => n.id === currentNovelId.value) || null
  )
  const novelsByStatus = computed(() => {
    const grouped: Record<string, Novel[]> = {}
    for (const novel of novels.value) {
      if (!grouped[novel.status]) grouped[novel.status] = []
      grouped[novel.status].push(novel)
    }
    return grouped
  })

  // Actions
  async function fetchNovels() {
    loading.value = true
    error.value = null
    try {
      const response = await api.get('/novels')
      const validated = z.array(NovelSchema).parse(response)
      novels.value = validated
    } catch (err) {
      error.value = err instanceof Error ? err.message : 'Unknown error'
      throw err
    } finally {
      loading.value = false
    }
  }

  async function createNovel(data: Partial<Novel>) {
    const response = await api.post('/novels', data)
    const novel = NovelSchema.parse(response)
    novels.value.unshift(novel)
    return novel
  }

  function setCurrentNovel(id: string) {
    currentNovelId.value = id
  }

  // WebSocket实时更新
  socket.on('novel:updated', (data: unknown) => {
    const update = NovelSchema.partial().parse(data)
    const index = novels.value.findIndex(n => n.id === update.id)
    if (index !== -1) {
      novels.value[index] = { ...novels.value[index], ...update }
    }
  })

  return {
    novels,
    currentNovelId,
    currentNovel,
    novelsByStatus,
    loading,
    error,
    fetchNovels,
    createNovel,
    setCurrentNovel,
  }
})

### 8.1 服务层架构

```typescript
// services/api.ts（重构版）
import axios, { AxiosInstance, AxiosRequestConfig } from 'axios'
import { z } from 'zod'
import { useAuthStore } from '@/stores/auth'

class ApiClient {
  private client: AxiosInstance
  private pendingRequests = new Map<string, AbortController>()

  constructor() {
    this.client = axios.create({
      baseURL: import.meta.env.VITE_API_BASE_URL || 'http://localhost:8004/api/v1',
      timeout: 60000,
      headers: { 'Content-Type': 'application/json' },
    })

    this.setupInterceptors()
  }

  private setupInterceptors() {
    // 请求拦截器: 认证 + 请求去重
    this.client.interceptors.request.use((config) => {
      const auth = useAuthStore()
      if (auth.token) {
        config.headers.Authorization = `Bearer ${auth.token}`
      }
      config.headers['X-Request-ID'] = crypto.randomUUID()

      // 请求去重
      const key = `${config.method}-${config.url}-${JSON.stringify(config.params)}`
      if (this.pendingRequests.has(key)) {
        this.pendingRequests.get(key)!.abort()
      }
      const controller = new AbortController()
      config.signal = controller.signal
      this.pendingRequests.set(key, controller)

      return config
    })

    // 响应拦截器: 统一错误处理 + 类型验证
    this.client.interceptors.response.use(
      (response) => {
        this.pendingRequests.delete(
          `${response.config.method}-${response.config.url}-${JSON.stringify(response.config.params)}`
        )
        return response.data
      },
      (error) => {
        if (axios.isCancel(error)) {
          return Promise.reject(new Error('Request cancelled'))
        }
        // 统一错误处理
        const status = error.response?.status
        const message = error.response?.data?.message || error.message
        if (status === 401) {
          useAuthStore().logout()
        }
        return Promise.reject(new ApiError(status, message))
      }
    )
  }

  // 类型安全的请求方法
  async get<T>(url: string, schema: z.ZodType<T>, config?: AxiosRequestConfig): Promise<T> {
    const data = await this.client.get(url, config)
    return schema.parse(data)
  }

  async post<T>(url: string, schema: z.ZodType<T>, body?: unknown, config?: AxiosRequestConfig): Promise<T> {
    const data = await this.client.post(url, body, config)
    return schema.parse(data)
  }
}

export class ApiError extends Error {
  constructor(public status: number, message: string) {
    super(message)
    this.name = 'ApiError'
  }
}

export const api = new ApiClient()
```

  private reconnectAttempts = 0
  private heartbeatTimer: ReturnType<typeof setInterval> | null = null
  private listeners = new Map<string, Set<(data: unknown) => void>>()

  public isConnected = ref(false)
  public latency = ref(0)

  constructor(private options: SocketOptions) {}

  connect() {
    this.ws = new WebSocket(this.options.url)

    this.ws.onopen = () => {
      this.isConnected.value = true
      this.reconnectAttempts = 0
      this.flushQueue()
      this.startHeartbeat()
    }

    this.ws.onmessage = (event) => {
      const message = JSON.parse(event.data)
      if (message.type === 'pong') {
        this.latency.value = Date.now() - message.timestamp
        return
      }
      this.emit(message.type, message.data)
    }

    this.ws.onclose = () => {
      this.isConnected.value = false
      this.stopHeartbeat()
      this.attemptReconnect()
    }

    this.ws.onerror = (error) => {
      console.error('WebSocket error:', error)
    }
  }

  private attemptReconnect() {
    const max = this.options.maxReconnectAttempts || 5
    if (this.reconnectAttempts >= max) return

    this.reconnectAttempts++
    setTimeout(() => this.connect(), this.options.reconnectInterval || 3000)
  }

  private startHeartbeat() {
    this.heartbeatTimer = setInterval(() => {
      this.send('ping', { timestamp: Date.now() })
    }, this.options.heartbeatInterval || 30000)
  }

  private stopHeartbeat() {
    if (this.heartbeatTimer) {
      clearInterval(this.heartbeatTimer)
      this.heartbeatTimer = null
    }
  }

  private flushQueue() {
    while (this.messageQueue.length > 0) {
      this.ws?.send(this.messageQueue.shift()!)
    }
  }

  send(type: string, data: unknown) {
    const message = JSON.stringify({ type, data })
    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.send(message)
    } else {
      this.messageQueue.push(message)
    }
  }

  on(event: string, callback: (data: unknown) => void) {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, new Set())
    }
    this.listeners.get(event)!.add(callback)

    return () => {
      this.listeners.get(event)?.delete(callback)
    }
  }

  private emit(event: string, data: unknown) {
    this.listeners.get(event)?.forEach(cb => cb(data))
  }

  disconnect() {
    this.stopHeartbeat()
    this.ws?.close()
  }
}

export const socket = new SocketClient({
  url: import.meta.env.VITE_WS_URL || 'ws://localhost:8004/ws',
  reconnectInterval: 3000,
  maxReconnectAttempts: 10,
  heartbeatInterval: 30000,
})
